Implementación de Grafo con Lista de Adyacencia (Python)

class Grafo:
    def __init__(self):
        "Inicializar grafo con lista de adyacencia"
        self.adyacencia = {}
    
    def agregar_vertice(self, vertice):
        "Agregar un vértice al grafo"
        if vertice not in self.adyacencia:
            self.adyacencia[vertice] = []
    
    def agregar_arista(self, v1, v2):
        "Agregar una arista entre dos vértices (no dirigida)"
        if v1 not in self.adyacencia:
            self.agregar_vertice(v1)
        if v2 not in self.adyacencia:
            self.agregar_vertice(v2)
        
        self.adyacencia[v1].append(v2)
        self.adyacencia[v2].append(v1)
    
    def obtener_vecinos(self, vertice):
        "Obtener los vecinos de un vértice"
        return self.adyacencia.get(vertice, [])

Algoritmo BFS (Python)

from collections import deque

def bfs(grafo, inicio):
    "Búsqueda en amplitud (BFS)"
    visitados = set()
    cola = deque([inicio])
    visitados.add(inicio)
    resultado = []
    
    while cola:
        vertice = cola.popleft()
        resultado.append(vertice)
        
        for vecino in grafo.obtener_vecinos(vertice):
            if vecino not in visitados:
                visitados.add(vecino)
                cola.append(vecino)
    
    return resultado

Algoritmo DFS (Python)

def dfs(grafo, inicio):
    "Búsqueda en profundidad (DFS)"
    visitados = set()
    resultado = []
    
    def _dfs_recursivo(vertice):
        visitados.add(vertice)
        resultado.append(vertice)
        
        for vecino in grafo.obtener_vecinos(vertice):
            if vecino not in visitados:
                _dfs_recursivo(vecino)
    
    _dfs_recursivo(inicio)
    return resultado

Matriz de Adyacencia (JavaScript)

class GrafoMatriz {
    constructor(numVertices) {
        this.numVertices = numVertices;
        this.matriz = Array(this.numVertices)
            .fill(null)
            .map(() => Array(this.numVertices).fill(0));
    }
    
    agregarArista(v1, v2) {
        // Para grafo no dirigido
        this.matriz[v1][v2] = 1;
        this.matriz[v2][v1] = 1;
    }
    
    tieneArista(v1, v2) {
        return this.matriz[v1][v2] === 1;
    }
}

BFS Iterativo (JavaScript)

function bfsIterativo(grafo, inicio) {
    const visitados = new Set();
    const cola = [inicio];
    const resultado = [];
    
    visitados.add(inicio);
    
    while (cola.length > 0) {
        const vertice = cola.shift();
        resultado.push(vertice);
        
        for (const vecino of grafo[vertice]) {
            if (!visitados.has(vecino)) {
                visitados.add(vecino);
                cola.push(vecino);
            }
        }
    }
    
    return resultado;
}

DFS Recursivo (JavaScript)

function dfsRecursivo(grafo, inicio) {
    const visitados = new Set();
    const resultado = [];
    
    function _dfs(vertice) {
        visitados.add(vertice);
        resultado.push(vertice);
        
        for (const vecino of grafo[vertice]) {
            if (!visitados.has(vecino)) {
                _dfs(vecino);
            }
        }
    }
    
    _dfs(inicio);
    return resultado;
}

Ejemplo de Uso

# Crear grafo
g = Grafo()
g.agregar_arista('A', 'B')
g.agregar_arista('A', 'C')
g.agregar_arista('B', 'D')
g.agregar_arista('C', 'E')

# Ejecutar BFS
print("BFS desde A:", bfs(g, 'A'))
# Output: ['A', 'B', 'C', 'D', 'E']

# Ejecutar DFS
print("DFS desde A:", dfs(g, 'A'))
# Output: ['A', 'B', 'D', 'C', 'E']
